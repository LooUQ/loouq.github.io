44  template <typename T>
45  struct ObjectReference
46  {
48  T * object = nullptr;
49 
51  unsigned int m_References = 0;
52 
54  ObjectReference<T>() {};
55 
58  ObjectReference<T>(T* ptrToObj) : object(ptrToObj) {}
59 
61  virtual ~ObjectReference<T>() { delete object; m_References = 0; }
62 
64  void operator--() {
65  m_References--;
66  if (m_References <= 0) {
67  delete this;
68  }
69  }
70 
72  void operator++() {
73  m_References++;
74  }
75 
78  unsigned int GetUses() const noexcept { return m_References; }
79  };
