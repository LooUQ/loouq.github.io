129  inline size_t TStream<T, N, ULBuffer>::Write(const T block[], size_t blocksz)
130  {
131  if (!blocksz) return NULL;
132  size_t result;
133  result = m_Buffer->PutN(block, blocksz);
134  m_Flags.eofbit = 0;
135  return result;
136  }

147  template<typename T, unsigned int N, typename ULBuffer>
148  inline size_t TStream<T, N, ULBuffer>::Read(T block[], size_t blocksz)
149  {
150  if (m_Flags.eofbit) return NULL;
151 
152  size_t result = m_Buffer->GetN(block, blocksz);
153  if (result == NULL)
154  {
155  m_Flags.eofbit = 1;
156  }
157  else
158  {
159  m_Flags.goodbit = 1;
160  }
161  return result;
162  }
